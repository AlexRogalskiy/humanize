Humanize JSF Example

Format

Source

<h:outputFormat value="hi {0}!">
 <f:param value="ho" />
</h:outputFormat>

<h:outputFormat value="hi {0,humanize,binary.prefix}!">
 <f:param value="\#{10}" />
</h:outputFormat>

Output

Plurals

Source

<h:outputText value="something::{0} things">
   <hmnz:pluralize value="1" />
</h:outputText>
<h:outputText value="no{1}::some{1}::{0} {1}s">
  <hmnz:pluralize value="2" args="thing" />
</h:outputText>
<h:outputText value="no{1}::some{1}::{0} {1}s">
  <hmnz:pluralize value="0" args="thing" />
</h:outputText>
<h:outputText value="\#{msgs['plural.template']}">
  <hmnz:pluralize value="100" />
</h:outputText>
<h:outputText value="\#{msgs['plural.template']}">
  <hmnz:pluralize value="\#{1}" />
</h:outputText>
